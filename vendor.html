let currentStep = 0; const formSteps = document.querySelectorAll(".form-step"); const steps = document.querySelectorAll(".step"); let actionType = ''; function updateProgressBar() { steps.forEach((step, index) => { if (index === currentStep) { step.classList.add("active"); } else { step.classList.remove("active"); } }); } function showForm(type) { actionType = type; nextStep(); } function nextStep() { formSteps[currentStep].classList.add("hidden"); if (actionType === 'register') { if (currentStep === 0) { // Show register form document.querySelector('.register-form').classList.remove("hidden"); } if (currentStep < 2) currentStep++; } else if (actionType === 'update') { if (currentStep === 0) { // Show update form step (Customer ID and Name) document.querySelector('.update-form').classList.remove("hidden"); } else if (currentStep === 1) { // Show full update details form document.querySelector('.update-details-form').classList.remove("hidden"); } if (currentStep < 2) currentStep++; } else if (actionType === 'delete') { if (currentStep === 0) { // Show delete form step (Customer ID and Name) document.querySelector('.update-form').classList.remove("hidden"); } else if (currentStep === 1) { // Show delete confirmation form document.querySelector('.delete-form').classList.remove("hidden"); } if (currentStep < 2) currentStep++; } updateProgressBar(); } function prevStep() { formSteps[currentStep].classList.add("hidden"); currentStep--; formSteps[currentStep].classList.remove("hidden"); updateProgressBar(); } document.getElementById("registerForm").addEventListener("submit", function (e) { e.preventDefault(); alert("Customer registered successfully!"); }); document.getElementById("actionForm").addEventListener("submit", function (e) { e.preventDefault(); if (actionType === 'delete') { alert("Customer deleted successfully!"); } else if (actionType === 'update') { alert("Customer updated successfully!"); } });
